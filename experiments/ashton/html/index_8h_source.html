59  // For each index format we forward the method call.
60  // We use C++11 variadic templates and perfect forwarding to implement foreach() generically,
61  // and split the types of the methods from the arguments passed to allow implicit conversions.
62  template<class... Ts,class... As>
63  void foreach(void (IndexIntf::*methodPtr)(Ts...),As&&... args)
64  {
65  QListIterator<IndexIntf> li(m_intfs);
66  for (li.toFirst();li.current();++li)
67  {
68  (li.current()->*methodPtr)(std::forward<As>(args)...);
69  }
70  }
71 
72  public:
74  IndexList() { m_intfs.setAutoDelete(TRUE); m_enabled=TRUE; }
76  void addIndex(IndexIntf *intf)
77  { m_intfs.append(intf); }
78  void disable()
79  { m_enabled = FALSE; }
80  void enable()
81  { m_enabled = TRUE; }
82  bool isEnabled() const
83  { return m_enabled; }
84 
85  // IndexIntf implementation
86  void initialize()
87  { foreach(&IndexIntf::initialize); }
88  void finalize()
89  { foreach(&IndexIntf::finalize); }
90  void incContentsDepth()
91  { if (m_enabled) foreach(&IndexIntf::incContentsDepth); }
92  void decContentsDepth()
93  { if (m_enabled) foreach(&IndexIntf::decContentsDepth); }
94  void addContentsItem(bool isDir, const char *name, const char *ref,
95  const char *file, const char *anchor,bool separateIndex=FALSE,bool addToNavIndex=FALSE,
96  const Definition *def=0)
97  { if (m_enabled) foreach(&IndexIntf::addContentsItem,isDir,name,ref,file,anchor,separateIndex,addToNavIndex,def); }
98  void addIndexItem(const Definition *context,const MemberDef *md,const char *sectionAnchor=0,const char *title=0)
99  { if (m_enabled) foreach(&IndexIntf::addIndexItem,context,md,sectionAnchor,title); }
100  void addIndexFile(const char *name)

102  void addImageFile(const char *name)

104  void addStyleSheetFile(const char *name)

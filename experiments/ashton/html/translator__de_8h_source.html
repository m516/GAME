755  virtual QCString trWriteList(int numEntries)
756  {
757  QCString result;
758  int i;
759  // the inherits list contain `numEntries' classes
760  for (i=0;i<numEntries;i++)
761  {
762  // use generateMarker to generate placeholders for the class links!
763  result+=generateMarker(i); // generate marker for entry i in the list
764  // (order is left to right)
765 
766  if (i!=numEntries-1) // not the last entry, so we need a separator
767  {
768  if (i<numEntries-2) // not the fore last entry
769  result+=", ";
770  else // the fore last entry
771  result+=" und ";
772  }
773  }
774  return result;
775  }

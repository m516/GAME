500  // For each output format that is enabled (OutputGenerator::isEnabled()) we forward
501  // the method call.
502  // We use C++11 variadic templates and perfect forwarding to implement forall() generically,
503  // and split the types of the methods from the arguments passed to allow implicit conversions.
504  template<typename T,class... Ts,class... As>
505  void forall(void (T::*methodPtr)(Ts...),As&&... args)
506  {
507  QListIterator<OutputGenerator> li(m_outputs);
508  OutputGenerator *og;
509  for (li.toFirst();(og=li.current());++li)
510  {
511  if (og->isEnabled()) (og->*methodPtr)(std::forward<As>(args)...);
512  }
513  }

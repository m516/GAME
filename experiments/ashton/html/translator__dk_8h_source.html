659  virtual QCString trWriteList(int numEntries)
660  {
661  QCString result;
662  int i;
663  // the inherits list contain `numEntries' classes
664  for (i=0;i<numEntries;i++) {
665  // use generateMarker to generate placeholders for the class links!
666  result+=generateMarker(i); // generate marker for entry i in the list
667  // (order is left to right)
668 
669  if (i!=numEntries-1) { // not the last entry, so we need a separator
670  if (i<numEntries-2) // not the fore last entry
671  result+=", ";
672  else // the fore last entry
673  result+=" og ";
674  }
675  }
676  return result;
677  }
